function	script	F_GetRDMOPT_Name	{
	switch(getarg(0)){
        case 3: return "Str";
        case 4: return "Agi";
        case 5: return "Vit";
        case 6: return "Int";
        case 7: return "Dex";
        case 8: return "Luk";
        case 9: return "Max HP";
        case 14: return "Magic Attack";
		case 17: return "ATK";
		case 18: return "HIT";
		case 19: return "MATK";
		case 20: return "DEF";
		case 22: return "FLEE";
        case 23: return "Perfect Dodge";
		case 24: return "CRIT";
		case 26: return "Water elem. resistance";
		case 27: return "Earth elem. resistance";
		case 28: return "Fire elem. resistance";
		case 29: return "Wind elem. resistance";
        case 32: return "Shadow elem. resistance";
        case 33: return "Ghost elem. resistance";
        case 34: return "Undead elem. resistance";
        case 94: return "Reduce dmg from DemiHuman";
        case 104: return "Increase dmg from DemiHuman";
        case 164: return "Critical Damage";
        case 166: return "Increase Long Range Atk";
		default: return "Unknow Enchant";
	}
}
