prontera,188,198,2	script	SG Resetter	10437 ,{
cutin "ex_se_happycloud03.png",2;
   mes "[sG Resetter]";
   mes "What do you wish to reset?";
   menu "Reset Feeling",feeling,"Reset Hatred",hatred;

feeling:
   atcommand "@feelreset";
   mes "Done.";
	cutin "",255;
                close;


hatred:
   set PC_HATE_MOB_MOON, 0;
   set PC_HATE_MOB_STAR, 0;
   set PC_HATE_MOB_SUN, 0;
   mes "Done.";
   next;
   mes "[sG Resetter]";
   mes "Hatred will be reset upon relogging.";
   mes "Do you wish to relog now?";
   menu "Relog now",relog,"Later",later;

relog:
   atcommand "@kick "+strcharinfo(0);
   end;

later:
   next;
   mes "[sG Resetter]";
   mes "Ok then, but don't forget changes won't take effect until you relog.";
	cutin "",255;
                close;


OnInit:
	while(1) {
		showscript "Feel Reset", getnpcid(0);
		sleep 1000;
	}
	end;

OnPCLoadMapEvent:
if( strcharinfo(3) == strnpcinfo(4) )
	showevent QTYPE_CLICKME;
end;
}
